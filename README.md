# react-textarea-markdown
###### markdown for react

---
### install module
```
npm i react-textarea-markdown
```
### for example
```
import React, { Component } from "react";
import Markdown from "react-textarea-markdown";
class App extends Component {
	render () {
		return (
			<div className="App">
				<Markdown textarea={true} />
			</div>
		);
	}
}
```
